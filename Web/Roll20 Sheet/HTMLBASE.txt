            setAttrs({
                [`${stat}_bonus`]: stat_bonus
            });
        });
    });
});

            var psbase = parseInt(values["ps_dice"],10) || 0;
            var pebase = parseInt(values["pe_dice"],10) || 0;

            var pebon = values.pe_extra;

            setAttrs({
                "ps_current": psmax,
                "pe_current": pemax,
                "ps_max": psmax,
                "pe_max": pemax
            });
            
        });
    });
